21 FunctionReturn< Function > executeOnMain( Function&& func, Args... args )
22 {
23 auto main { mainThread() };
24 if ( main == nullptr )
25 throw std::runtime_error( "Tried to run executeOnMain without setting a main thread first!" );
26
27 if ( QThread::currentThread() == mainThread() )
28 {
29 //Execute as normal
30 func( args... );
31 }
32 else
33 {
34 QTimer timer;
35 timer.moveToThread( main );
36 //As long as we move the timer to the main thread it'll execute it's timeout on the main thread.
37 timer.setSingleShot( true );
38
39 using RetType = FunctionDecomp< Function >::ResultType;
40
41 std::promise< RetType > promise {};
42 std::future< RetType > future { promise.get_future() };
43
44 std::tuple< Args... > args_tpl { args... };
45
46 QObject::connect(
47 &timer,
48 &QTimer::timeout,
49 //Safe to ref args here since we copy them into our function AND block on the future for completion
50 [ &promise, func, &args_tpl ]()
51 {
52 if constexpr ( std::same_as< RetType, void > )
53 {
54 std::apply( func, args_tpl );
55 promise.set_value();
56 }
57 else
58 {
59 promise.set_value( std::apply( func, args_tpl ) );
60 }
61 } );
62
63 QMetaObject::invokeMethod( &timer, "start", Qt::QueuedConnection, Q_ARG( int, 0 ) );
64
65 //Wait until the main thread executes what we asked.
