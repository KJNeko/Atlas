48template < typename T >
49inline T getSettings( const QString setting_name )
50{
51 QSettings settings { getSettingsObject() };
52 const auto variant { settings.value( setting_name ) };
53 if ( variant.template canConvert< T >() )
54 return variant.template value< T >();

56 {
57 spdlog::warn( "Setting for {} was not populated!", setting_name );

60 return {};
61 else
62 throw std::runtime_error(
63 "T was not default constructable! Throwing instead! For given setting:" + setting_name.toStdString() );
